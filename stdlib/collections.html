jade.stdlib.collections

dynamic collections — List, Map, Set.

import jade.stdlib.collections;
// or import specific ones:
import jade.stdlib.collections.{ List, Map, Set };

List\<T\>

a dynamic array that grows as you add items.

var list = List<i32>.new();
list.push(1);
list.push(2);
list.push(3);

methods

list.push(item T)              // add to end
list.pop() T?                  // remove and return last item
list.get(i i32) T              // get item at index
list.set(i i32, val T)         // set item at index
list.insert(i i32, val T)      // insert at index
list.removeAt(i i32)           // remove at index
list.len() i32                 // number of items
list.isEmpty() bool
list.first() T?                // first item or null
list.last() T?                 // last item or null
list.clear()                   // remove all items
var nums = List<i32>.new();
nums.push(10);
nums.push(20);
nums.push(30);

io.println("#{nums.len()}");       // 3
io.println("#{nums.get(1)}");      // 20
io.println("#{nums.last()}");      // 30

nums.removeAt(0);
io.println("#{nums.get(0)}");      // 20

for i in 0..nums.len() {
    io.println("#{nums.get(i)}");
}

Map\<K, V\>

a key-value store. keys are strings.

var scores = Map<str, i32>.new();
scores.set("Alice", 95);
scores.set("Bob",   87);
scores.set("Carol", 92);

methods

map.set(key str, val V)        // set a key
map.get(key str) V?            // get value or null
map.has(key str) bool          // check if key exists
map.delete(key str)            // remove a key
map.len() i32                  // number of entries
map.clear()
var ages = Map<str, i32>.new();
ages.set("Alice", 30);
ages.set("Bob",   25);

if ages.has("Alice") {
    io.println("Alice is #{ages.get("Alice")}");
}

ages.delete("Bob");
io.println("#{ages.len()} people");  // 1

Set\<T\>

a collection of unique string values.

var tags = Set<str>.new();
tags.add("jade");
tags.add("programming");
tags.add("jade");  // duplicate — ignored
io.println("#{tags.len()}");  // 2

methods

set.add(val str)               // add a value
set.has(val str) bool          // check membership
set.remove(val str)            // remove a value
set.len() i32
set.isEmpty() bool
set.clear()
var fruits = Set<str>.new();
fruits.add("apple");
fruits.add("banana");
fruits.add("cherry");

io.println("has apple: #{fruits.has("apple")}");    // true
io.println("has grape: #{fruits.has("grape")}");    // false

fruits.remove("banana");
io.println("#{fruits.len()} fruits");  // 2

example: word counter

import jade.stdlib.collections.{ Map };
import jade.stdlib.io;

@entry
def main() i32 {
    val words = ["jade", "is", "cool", "jade", "is", "fast", "jade"];
    var counts = Map<str, i32>.new();

    for word in words {
        val current = counts.get(word);
        if current == null {
            counts.set(word, 1);
        } else {
            counts.set(word, current + 1);
        }
    }

    io.println("jade appears #{counts.get("jade")} times");  // 3
    return 0;
}
