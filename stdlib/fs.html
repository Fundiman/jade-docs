jade.stdlib.fs

filesystem operations.

import jade.stdlib.fs;

reading and writing

fs.read(path str) Result<str>

read entire file as a string.

val content = fs.read("config.json");
when content {
    is Ok(s)  -> io.println("file: #{s}");
    is Err(e) -> io.println("error: #{e}");
}

fs.write(path str, content str) Result<void>

write a string to a file. creates or overwrites.

val result = fs.write("output.txt", "hello Jade!");
when result {
    is Ok(_)  -> io.println("written!");
    is Err(e) -> io.println("write failed: #{e}");
}

fs.append(path str, content str) Result<void>

append to a file without overwriting.

fs.append("log.txt", "new entry\n");

file information

fs.exists(path str) bool

if fs.exists("config.json") {
    io.println("config found");
}

fs.isDir(path str) bool

if fs.isDir("/tmp") {
    io.println("is a directory");
}

fs.size(path str) i64

file size in bytes. returns -1 on error.

val bytes = fs.size("myfile.txt");
io.println("size: #{bytes} bytes");

file operations

fs.delete(path str) Result<void>

fs.delete("temp.txt");

fs.copy(src str, dst str) Result<void>

fs.copy("original.txt", "backup.txt");

fs.mkdir(path str) Result<void>

create a directory.

fs.mkdir("output/reports");

example: reading a config file

import jade.stdlib.fs;
import jade.stdlib.io;

def loadConfig() Result<str> {
    if !fs.exists("config.json") {
        return Result.Err("config.json not found");
    }
    return fs.read("config.json");
}

@entry
def main() i32 {
    val config = loadConfig();
    when config {
        is Ok(s)  -> io.println("loaded config: #{s.len()} bytes");
        is Err(e) -> io.println("#{e}");
    }
    return 0;
}
