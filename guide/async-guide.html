async / await

the basics

mark a function async to make it asynchronous. use .await() to wait for the result:

async def fetchUser(id u32) Result<User> {
    val response = http.get("https://api.example.com/users/#{id}").await();
    when response {
        is Ok(data)  -> return Result.Ok(User.fromJson(data));
        is Err(msg)  -> return Result.Err(msg);
    }
}

@entry
async def main() {
    val user = fetchUser(42).await();
    when user {
        is Ok(u)  -> io.println("hello #{u.name}");
        is Err(e) -> io.println("error: #{e}");
    }
}

why async?

some operations take time — network requests, file I/O, database queries. without async, your program would freeze while waiting. with async, other work can happen while you wait.

awaiting multiple things

run things in parallel:

async def main() {
    val (user, posts) = parallel(
        fetchUser(42),
        fetchPosts(42)
    ).await();
}

or sequentially:

async def main() {
    val user  = fetchUser(42).await();
    val posts = fetchPosts(user.id).await();
}

async in classes

class ApiClient {
    val base_url str

    async def get(path str) Result<str> {
        return http.get("#{base_url}#{path}").await();
    }

    async def post(path str, body str) Result<str> {
        return http.post("#{base_url}#{path}", body).await();
    }
}

error handling with async

async functions return Result<T> just like regular functions:

async def loadConfig(path str) Result<Config> {
    val content = fs.read(path).await();
    when content {
        is Err(e) -> return Result.Err("failed to read config: #{e}");
        is Ok(s)  -> return Config.parse(s);
    }
}

timeouts

async def fetchWithTimeout(url str, ms u64) Result<str> {
    val result = race(
        http.get(url),
        time.sleep(ms).map(def() Result<str> = Result.Err("timeout"))
    ).await();
    return result;
}

current status

async/await syntax is fully parsed and type-checked by jscc. the runtime scheduler in jvk currently runs async tasks sequentially — true concurrent suspension/resumption is on the roadmap for v0.3.0-alpha.
